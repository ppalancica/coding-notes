class Solution {
    func lengthOfLastWord(_ s: String) -> Int {
        if s.isEmpty { return 0 }
        
        let array = Array(s)
        var length = s.count
        var lastWordLength = 0

        while length > 0 && array[length - 1] == " " {
            length -= 1
        }
            
        while length > 0 && array[length - 1] != " " {
            lastWordLength += 1
            length -= 1
        }
            
        return lastWordLength
    }
}
