--- Stack of Int items

class Stack {
    var items: [Int] = []
    
    func push(n: Int) {
        items.append(n)
    }
    
    func pop() -> Int? {
        return items.popLast()
    }

    func peek() -> Int? {
        return items.last
    }
}

--- Queue of Int items

class Queue {
    var items: [Int] = []
    
    func enqueue(n: Int) {
        items.append(n)
    }
    
    func dequeue() -> Int? {
        return items.removeFirst()
    }

    func peak() -> Int? {
        return items.first
    }
}
